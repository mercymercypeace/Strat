-- Services
local TeleportService = game:GetService("TeleportService")
local GuiService = game:GetService("GuiService")
local Players = game:GetService("Players")

-- Settings
local RejoinDelay = 5 

-- Function to Rejoin
local function Rejoin()
    local player = Players.LocalPlayer
    -- Teleporting using only PlaceId will find a new public server instance
    -- instead of trying to rejoin the specific JobId (the specific room)
    TeleportService:Teleport(game.PlaceId, player)
end

-- Monitor for Disconnect UI (Error messages)
GuiService.ErrorMessageChanged:Connect(function()
    print("Connection issue detected. Rejoining in " .. RejoinDelay .. " seconds...")
    task.wait(RejoinDelay)
    Rejoin()
end)

-- Backup: Periodic UI scan for error prompts
task.spawn(function()
    while task.wait(10) do
        local coreGui = game:GetService("CoreGui")
        local promptGui = coreGui:FindFirstChild("RobloxPromptGui")
        if promptGui then
            local prompt = promptGui:FindFirstChild("promptOverlay")
            if prompt and prompt:FindFirstChild("ErrorPrompt") then
                print("Error prompt found. Rejoining to a new session...")
                Rejoin()
            end
        end
    end
end)

print("Auto-Rejoin (Matchmaking Mode) is active.")
